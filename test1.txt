    public function tambahbackup(Request $request){
        $this->validate($request, [
            'nama_event' => 'required',
            'ketegori_event' => 'required',
            'keterangan_tiket[]' => 'required',
            'harga_tiket[]' => 'required',
            'jumlah_tiket[]' => 'required',
            'waktu_mulai_event' => 'required',
            'waktu_berakhir_event' => 'required',
            'lokasi_event' => 'required',
            'deskripsi_event' => 'required',
            'nama_penyelenggara' => 'required',
            'no_telp_penyelenggara' => 'required',
            'no_rek_penyelenggara' => 'required',
            'nama_bank_penyelenggara' => 'required'
        ]);

        //Event
        $event = new Event();
        $event->nama_event = $request->input('nama_event');
        $event->ketegori_event = $request->input('ketegori_event');
        $event->waktu_mulai_event = $request->input('waktu_mulai_event');
        $event->waktu_berakhir_event = $request->input('waktu_berakhir_event');
        $event->lokasi_event = $request->input('lokasi_event');
        $event->deskripsi_event = $request->input('deskripsi_event');
        $event->save();

        //Penyelenggara
        $penyelenggara = new Penyelenggara();
        $penyelenggara->nama_penyelenggara = $request->input('nama_penyelenggara');
        $penyelenggara->no_telp_penyelenggara = $request->input('no_telp_penyelenggara');
        $penyelenggara->no_rek_penyelenggara = $request->input('no_rek_penyelenggara');
        $penyelenggara->nama_bank_penyelenggara = $request->input('nama_bank_penyelenggara');

        //Tiket
        $nama_tiket = $request->input('nama_tiket');
        $keterangan_tiket = $request->input('keterangan_tiket');
        $harga_tiket = $request->input('harga_tiket');
        $jumlah_tiket = $request->input('jumlah_tiket');
        $count = count($nama_tiket);

        for ($i = 0; $i < $count; $i++) {
            if (isset($nama_tiket[$i])) {  //for check value is set or not..
                $tiket = new Tiket();
                $tiket->nama_tiket = $nama_tiket[$i]; //different number
                $tiket->keterangan_tiket = $keterangan_tiket[$i];
                $tiket->harga_tiket = $harga_tiket[$i];
                $tiket->jumlah_tiket = $jumlah_tiket[$i];
                $tiket->save();
            }
        }

        return redirect('http://google.com');
    }
